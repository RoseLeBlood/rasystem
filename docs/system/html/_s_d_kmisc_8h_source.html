220 // Simple helper stack class
221 //--------------------------------------------------------------------------------------
222 template <class T> class CDXUTStack
223 {
224 private:
225  UINT m_MemorySize;
226  UINT m_NumElements;
227  T* m_pData;
228 
229  bool EnsureStackSize( UINT64 iElements )
230  {
231  if( m_MemorySize > iElements )
232  return true;
233 
234  T* pTemp = new T[ ( size_t )( iElements * 2 + 256 ) ];
235  if( !pTemp )
236  return false;
237 
238  if( m_NumElements )
239  {
240  CopyMemory( pTemp, m_pData, ( size_t )( m_NumElements * sizeof( T ) ) );
241  }
242 
243  if( m_pData ) delete []m_pData;
244  m_pData = pTemp;
245  return true;
246  }
247 
248 public:
249  CDXUTStack()
250  {
251  m_pData = NULL; m_NumElements = 0; m_MemorySize = 0;
252  }
253  ~CDXUTStack()
254  {
255  if( m_pData ) delete []m_pData;
256  }
257 
258  UINT GetCount()
259  {
260  return m_NumElements;
261  }
262  T GetAt( UINT i )
263  {
264  return m_pData[i];
265  }
266  T GetTop()
267  {
268  if( m_NumElements < 1 )
269  return NULL;
270 
271  return m_pData[ m_NumElements - 1 ];
272  }
273 
274  T GetRelative( INT i )
275  {
276  INT64 iVal = m_NumElements - 1 + i;
277  if( iVal < 0 )
278  return NULL;
279  return m_pData[ iVal ];
280  }
281 
282  bool Push( T pElem )
283  {
284  if( !EnsureStackSize( m_NumElements + 1 ) )
285  return false;
286 
287  m_pData[m_NumElements] = pElem;
288  m_NumElements++;
289 
290  return true;
291  }
292 
293  T Pop()
294  {
295  if( m_NumElements < 1 )
296  return NULL;
297 
298  m_NumElements --;
299  return m_pData[m_NumElements];
300  }
301 };
